cmake_minimum_required(VERSION 3.12)
project (OTIS3LABA)

find_package(Python REQUIRED COMPONENTS Interpreter)

# Добавьте команды для каждого из ваших Python-скриптов
add_custom_target(run_alg_button_lid.py ALL
  COMMAND ${Python_EXECUTABLE} ${CMAKE_CURRENT_SOURCE_DIR}/alg_button_lid.py
  COMMENT "Running alg_button_lid.py"
)

add_custom_target(run_export.py ALL
  COMMAND ${Python_EXECUTABLE} ${CMAKE_CURRENT_SOURCE_DIR}/export.py
  COMMENT "Running export.py"
)

add_custom_target(run_lib.py ALL
  COMMAND ${Python_EXECUTABLE} ${CMAKE_CURRENT_SOURCE_DIR}/lib.py
  COMMENT "Running lib.py"
)
